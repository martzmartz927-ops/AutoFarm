-- ТАБЛИЦА ИГРОКОВ ИЗ СЕТЕВЫХ ПАКЕТОВ (100% надёжный способ,
-- работает на CRMP/Arizona, где sampGetPlayerIdByCharHandle и
-- sampGetCharHandleBySampPlayerId сломаны/крашат)
------------------------------------------------------------
-- ФИКС (важно): предыдущий подход пытался сопоставить playerId с ped-хэндлом
-- через onPlayerStreamIn (позиция на момент появления в стриме) — но
-- проверка показала, что onPlayerStreamIn почти не срабатывает на Arizona
-- (join=547, in=24 — то есть только 24 из 547 игроков вообще получили
-- стрим-ивент). Прямой опкод sampGetPlayerIdByCharHandle тоже не работает
-- (подтверждено тестовой кнопкой — срабатывает только для своего PLAYER_PED).
--
-- Правильное решение: НЕ ИСКАТЬ ped-хэндл вообще. onPlayerSync (и
-- onFootSync/onVehicleSync в некоторых версиях SAMP.Lua) шлёт playerId +
-- реальную позицию + health НАПРЯМУЮ из сети, регулярно (каждые несколько
-- тиков движения), а не один раз при появлении. Просто храним последнюю
-- известную позицию и health для каждого playerId — этого достаточно для
-- ESP, ped-хэндл не нужен вообще (координаты передаются серверу и так,
-- используем их напрямую для world->screen).
local samp_players = {} -- [playerId] = {nickname=, isNpc=, x=, y=, z=, health=, lastUpdate=}

-- счётчики срабатывания событий с момента запуска скрипта — диагностика
local event_stats = { playerJoinFired = 0, playerSyncFired = 0, footSyncFired = 0 }

function sampev.onPlayerJoin(playerId, color, isNpc, nickname)
    event_stats.playerJoinFired = event_stats.playerJoinFired + 1
    samp_players[playerId] = samp_players[playerId] or {}
    samp_players[playerId].nickname = nickname
    samp_players[playerId].isNpc = isNpc
end

function sampev.onPlayerQuit(playerId, reason)
    samp_players[playerId] = nil
end

-- Универсальный разборщик: на разных сборках SAMP.Lua/компат-профилях (например,
-- ARZMOD на MonetLoader/Android) порядок и число аргументов в этих колбэках может
-- отличаться от оригинальной THE-FYP/SAMP.Lua. Чтобы не зависеть от порядка,
-- сами ищем среди переданных аргументов: playerId (число) и data (таблица).
local function extract_id_and_data(...)
    local n = select('#', ...)
    local id, data
    for i = 1, n do
        local v = select(i, ...)
        if type(v) == 'table' and not data then
            data = v
        elseif type(v) == 'number' and not id then
            id = v
        end
    end
    return id, data
end

-- ОСНОВНОЙ источник координат: приходит регулярно, пока игрок пеший.
function sampev.onPlayerSync(...)
    local playerId, data = extract_id_and_data(...)
    if not playerId or not data then return end

    event_stats.playerSyncFired = event_stats.playerSyncFired + 1
    samp_players[playerId] = samp_players[playerId] or {}
    local p = samp_players[playerId]
    if data.position then
        p.x, p.y, p.z = data.position.x, data.position.y, data.position.z
    end
    if data.health then p.health = data.health end
    p.lastUpdate = os.clock()
    p.inVehicle = false
end

-- Некоторые сборки SAMP.Lua называют это onFootSync вместо onPlayerSync —
-- регистрируем оба на случай различий в версии библиотеки на этом сервере.
if sampev.onFootSync == nil then
    function sampev.onFootSync(...)
        return sampev.onPlayerSync(...)
    end
end

-- Пассажир/водитель в машине — синхронизация приходит другим пакетом,
-- у него тоже есть position. Обрабатываем, чтобы не терять игроков в тачках.
function sampev.onVehicleSync(...)
    local n = select('#', ...)
    local playerId, vehicleId, data
    local numbersFound = 0
    for i = 1, n do
        local v = select(i, ...)
        if type(v) == 'table' and not data then
            data = v
        elseif type(v) == 'number' then
            numbersFound = numbersFound + 1
            if numbersFound == 1 then playerId = v
            elseif numbersFound == 2 then vehicleId = v end
        end
    end
    if not playerId or not data then return end

    samp_players[playerId] = samp_players[playerId] or {}
    local p = samp_players[playerId]
    if data.position then
        p.x, p.y, p.z = data.position.x, data.position.y, data.position.z
    end
    if data.playerHealth then p.health = data.playerHealth end
    if data.health then p.health = data.health end
    p.vehicleId = vehicleId
    p.lastUpdate = os.clock()
    p.inVehicle = true
end

------------------------------------------------------------
-- РАЗДЕЛ PLAYER: TELEPORT TO WAYPOINT
------------------------------------------------------------
-- мгновенный телепорт персонажа к точке, отмеченной маркером (waypoint)
-- на игровой карте — работает как обычный маркер GPS, просто ставите
-- точку на карте и жмёте кнопку
local teleport_last_status = ''
local teleport_waypoint_request = nil -- {x, y, z} либо nil

-- Мгновенный скачок координат, но с попыткой скрыть сам факт скачка от
-- сервера: на короткое время приостанавливаем отправку пакетов позиции
-- (sampSetSendPacketsState), делаем прыжок, ждём немного и возобновляем
-- синхронизацию — сервер получает уже новую точку, без промежуточных
-- координат полёта.
--
-- ВАЖНО: это снижает риск детекта "по скорости/траектории движения", но
-- НЕ гарантирует защиту от детекта "по расстоянию между двумя соседними
-- принятыми позициями" — если анти-чит сверяет абсолютную дистанцию между
-- последней известной серверу точкой и новой, скачок всё равно будет виден
-- в момент возобновления синка. 100%-й гарантии от кика нет.
--
-- sampSetSendPacketsState — стандартная функция SAMP.Lua/MoonLoader для
-- управления отправкой исходящих пакетов; оборачиваем в pcall на случай,
-- если конкретная сборка её не поддерживает — тогда просто выполняем
-- обычный мгновенный телепорт без сокрытия.
local TELEPORT_HIDE_PAUSE_MS = 300 -- насколько приостанавливаем синк вокруг прыжка

local function teleport_in_jumps(px, py, pz, tx, ty, tz)
    local okPauseOff = pcall(sampSetSendPacketsState, false)

    pcall(setCharCoordinates, PLAYER_PED, tx, ty, tz)
    wait(TELEPORT_HIDE_PAUSE_MS)

    if okPauseOff then
        pcall(sampSetSendPacketsState, true)
    end
end

lua_thread.create(function()
    while true do
        wait(0)
        if teleport_waypoint_request then
            local target = teleport_waypoint_request
            teleport_waypoint_request = nil

            local bx, by, bz = target.x, target.y, target.z
            pcall(requestCollision, bx, by)
            pcall(loadScene, bx, by, bz)

            local finalZ = bz
            local okGz, gz = pcall(getGroundZFor3dCoord, bx, by, bz + 5)
            if okGz and type(gz) == 'number' and gz > -100 then
                finalZ = gz + 1.0
            end

            local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
            if okP and type(px) == 'number' then
                teleport_in_jumps(px, py, pz, bx, by, finalZ)
                teleport_last_status = u8'Телепорт выполнен.'
            else
                teleport_last_status = u8'Не удалось телепортироваться.'
            end
        end
    end
end)

local function bufToStr(buf, maxlen)
    local ok, s = pcall(ffi.string, buf, maxlen)
    if not ok then return '' end
    -- обрезаем по первому нулевому байту
    local z = s:find('\0')
    return z and s:sub(1, z - 1) or s
end

------------------------------------------------------------
-- AUTO RECONNECT: авто-ввод пароля при появлении диалога авторизации
------------------------------------------------------------

function sampev.onShowDialog(id, style, title, button1, button2, text)
    if not player_autoreconnect_enabled[0] then return end

    local pass_str = bufToStr(player_reconnect_password, 64)
    if pass_str == '' then return end

    -- ищем диалог авторизации по ключевым словам в заголовке/тексте —
    -- разные RP-серверы называют его по-разному, поэтому проверяем оба поля.
    local haystack = (tostring(title or '') .. ' ' .. tostring(text or '')):lower()
    local looks_like_auth =
        haystack:find('автори') or haystack:find('парол') or
        haystack:find('password') or haystack:find('login') or
        haystack:find('вход в аккаунт')

    if looks_like_auth then
        sampSendDialogResponse(id, 1, 65535, pass_str)
    end
end

local function add_object_to_list()
    local model_str = bufToStr(new_object_model, 16)
    local model_num = tonumber(model_str)
    if not model_num then return end -- некорректный ввод, ничего не делаем

    local name_str = bufToStr(new_object_name, 32)
    if name_str == '' then name_str = nil end

    table.insert(esp_objects_list, { model = math.floor(model_num), name = name_str })

    -- очищаем поля ввода после добавления
    new_object_model[0] = 0
    new_object_name[0] = 0
end

local function remove_object_from_list(index)
    table.remove(esp_objects_list, index)
end

-- есть ли modelId в списке ESP-объектов
local function is_esp_model(model)
    for _, v in ipairs(esp_objects_list) do
        if v.model == model then return true end
    end
    return false
end

lua_thread.create(function()
    while true do
        wait(300)

        if esp_enabled[0] and #esp_objects_list > 0 then
            local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
            if okP and type(px) == 'number' then
                local radius = esp_distance[0]
                local tmp = {}

                local okObjs, objs = pcall(getAllObjects)
                if okObjs and objs then
                    for _, handle in ipairs(objs) do
                        local okModel, model = pcall(getObjectModel, handle)
                        if okModel and model and is_esp_model(model) then
                            local okCoord, _, ox, oy, oz = pcall(getObjectCoordinates, handle)
                            if okCoord and type(ox) == 'number' then
                                local dist = getDistanceBetweenCoords3d(px, py, pz, ox, oy, oz)
                                if dist <= radius then
                                    tmp[#tmp + 1] = { handle = handle, x = ox, y = oy, z = oz, dist = dist }
                                end
                            end
                        end
                    end
                end

                esp_object_cache = tmp
            end
        else
            esp_object_cache = {}
        end
    end
end)

-- Поток: игроки
lua_thread.create(function()
    while true do
        wait(300)
        if esp_players_enabled[0] then
            local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
            if okP and type(px) == 'number' then
                local radius = esp_players_distance[0]
                local tmp = {}

                local dbgFound = 0

                if esp_players_nofilter[0] then
                    -- РЕЖИМ БЕЗ ФИЛЬТРА: никакой проверки игрок/NPC, просто все
                    -- существующие персонажи в радиусе. Нужен только чтобы
                    -- подтвердить, что сама отрисовка (world->screen) работает.
                    local allChars = getAllChars()
                    if allChars then
                        for _, ped in ipairs(allChars) do
                            if ped and ped ~= PLAYER_PED and doesCharExist(ped) then
                                local cx, cy, cz = getCharCoordinates(ped)
                                if cx then
                                    local dist = getDistanceBetweenCoords3d(px, py, pz, cx, cy, cz)
                                    if dist <= radius then
                                        dbgFound = dbgFound + 1
                                        tmp[#tmp + 1] = {
                                            x = cx, y = cy, z = cz, dist = dist,
                                            name = 'CHAR#' .. tostring(ped)
                                        }
                                    end
                                end
                            end
                        end
                    end

                    if esp_players_debug then
                        esp_players_debug = false
                        sampAddChatMessage(string.format(
                            '{AA88FF}[AXION DEBUG NOFILTER] chars=%d found=%d',
                            allChars and #allChars or 0, dbgFound), -1)
                    end

                else
                    -- ОСНОВНОЙ РЕЖИМ: читаем координаты напрямую из сетевых
                    -- пакетов (onPlayerSync/onVehicleSync), без поиска
                    -- ped-хэндла вообще. Это надёжно независимо от того,
                    -- работают ли на сервере опкоды ped<->id (подтверждено:
                    -- не работают) и независимо от частоты стрим-событий
                    -- (тоже подтверждено: onPlayerStreamIn почти не летит).
                    local now = os.clock()
                    local dbgTotal, dbgNpc, dbgStale, dbgNoPos = 0, 0, 0, 0

                    for pid, p in pairs(samp_players) do
                        dbgTotal = dbgTotal + 1
                        if p.isNpc then
                            dbgNpc = dbgNpc + 1
                        elseif not p.x then
                            dbgNoPos = dbgNoPos + 1
                        elseif p.lastUpdate and (now - p.lastUpdate) > STALE_TIMEOUT then
                            dbgStale = dbgStale + 1
                        else
                            local dist = getDistanceBetweenCoords3d(px, py, pz, p.x, p.y, p.z)
                            if dist <= radius then
                                dbgFound = dbgFound + 1
                                tmp[#tmp + 1] = {
                                    x = p.x, y = p.y, z = p.z, dist = dist,
                                    hp = p.health,
                                    name = p.nickname or ('ID:' .. tostring(pid))

-- ПОТОК: AUTO RECONNECT
------------------------------------------------------------
-- Следит, подключены ли мы к серверу. Если авто-реконнект включён и мы были
-- подключены хотя бы раз, а потом соединение пропало — эмулируем настоящий
-- обрыв связи через низкоуровневые raknet-пакеты (именно так это надёжно
-- работает на MonetLoader for Android — sampConnectToServer там ненадёжен).
-- Пароль вводится автоматически через onShowDialog, когда сервер покажет
-- диалог авторизации после того, как игра сама переподключится.
local sf_ok, sf = pcall(require, 'sampfuncs')
if not sf_ok and sampAddChatMessage then
    pcall(sampAddChatMessage, '{FF6347}[AXION] AUTO RECONNECT: модуль sampfuncs не найден, функция не будет работать.', -1)
end

local function do_reconnect_sequence()
    lua_thread.create(function()
        if not sf_ok or not sf then return end

        -- сначала аккуратно уведомляем сервер об отключении,
        -- затем эмулируем разрыв соединения — это заставляет клиент сам
        -- инициировать переподключение к последнему серверу (как при обычном
        -- обрыве интернета), включая повторный показ диалога авторизации.
        local okBs1, bs1 = pcall(raknetNewBitStream)
        if okBs1 and bs1 then
            pcall(raknetBitStreamWriteInt8, bs1, sf.PACKET_DISCONNECTION_NOTIFICATION)
            pcall(raknetSendBitStreamEx, bs1, sf.SYSTEM_PRIORITY, sf.RELIABLE, 0)
            pcall(raknetDeleteBitStream, bs1)
        end

        wait(500)

        local okBs2, bs2 = pcall(raknetNewBitStream)
        if okBs2 and bs2 then
            pcall(raknetEmulPacketReceiveBitStream, sf.PACKET_CONNECTION_LOST, bs2)
            pcall(raknetDeleteBitStream, bs2)
        end
    end)
end

lua_thread.create(function()
    while true do
        wait(500)

        local okAvail, available = pcall(isSampAvailable)
        local currentlyConnected = okAvail and available

        if currentlyConnected then
            player_was_connected_once = true
        elseif player_autoreconnect_enabled[0] and player_was_connected_once then
            local now = os.clock()
            if now - player_last_reconnect_attempt >= RECONNECT_RETRY_DELAY then
                player_last_reconnect_attempt = now
                do_reconnect_sequence()
            end
        end
    end
end)

-- ПОТОК: SPEED HACK
------------------------------------------------------------
-- Ускоряем скорость проигрывания анимаций ходьбы/бега/спринта напрямую —
-- этот способ не зависит от коллизии/трения текущей поверхности (асфальт,
-- трава, песок, уклон), в отличие от инжекта через setCharVelocity, который
-- на некоторых поверхностях гасится физическим движком раньше, чем успевает
-- подействовать (эффект "то тормозит, то без разницы").
-- За рулём анимации не участвуют в скорости движения — там ускоряем как раньше.
local SPEEDHACK_ANIMS = {
    'walk_player', 'walk_civi', 'walk_armed', 'walk_fat', 'walk_old', 'walk_fatold',
    'walk_gang1', 'walk_gang2', 'walk_drunk', 'walk_wuzi', 'walk_csaw', 'walk_rocket',
    'run_player', 'run_civi', 'run_armed', 'run_fat', 'run_fatold', 'run_old',
    'run_gang1', 'run_wuzi', 'run_csaw', 'run_rocket', 'run_1armed', 'run_left', 'run_right',
    'sprint_civi', 'sprint_panic', 'sprint_wuzi', 'swat_run',
    'woman_walknorm', 'woman_walkbusy', 'woman_walkpro', 'woman_walksexy',
    'woman_walkshop', 'woman_walkfatold', 'woman_walkold',
    'woman_run', 'woman_runbusy', 'woman_runfatold', 'woman_runpanic', 'woman_runsexy',
    'jog_malea', 'jog_femalea',
}

local SPEEDHACK_MAX_STEP = 0.6 -- максимальное доп. смещение за один тик (для транспорта, как раньше)

lua_thread.create(function()
    local prev_pos = nil
    local anims_applied = false

    while true do
        wait(0)

        local okVeh, veh = pcall(getCharVehicle, PLAYER_PED)
        local inVehicle = okVeh and veh and veh ~= 0

        if player_speedhack_enabled[0] then
            if not inVehicle then
                -- пешком: ускоряем анимации напрямую, каждый кадр (значение
                -- может быть изменено ползунком "на лету")
                local mult = player_speedhack_mult[0]
                for _, anim in ipairs(SPEEDHACK_ANIMS) do
                    pcall(setCharAnimSpeed, PLAYER_PED, anim, mult)
                end
                anims_applied = true
            else
                -- за рулём анимации ни при чём — используем прежний подход
                -- через усиление реального смещения между кадрами
                local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
                if okP and type(px) == 'number' then
                    if prev_pos then
                        local dx, dy = px - prev_pos.x, py - prev_pos.y
                        local horizDist = math.sqrt(dx*dx + dy*dy)
                        if horizDist > 0.005 and horizDist < 2.0 then
                            local mult = player_speedhack_mult[0]
                            local addDist = horizDist * mult
                            if addDist > SPEEDHACK_MAX_STEP then
                                addDist = SPEEDHACK_MAX_STEP
                            end
                            local nx = dx / horizDist * addDist
                            local ny = dy / horizDist * addDist
                            local vz = 0
                            local okVel, vvx, vvy, vvz = pcall(getCharVelocity, PLAYER_PED)
                            if okVel and type(vvz) == 'number' then vz = vvz end
                            pcall(setCharVelocity, PLAYER_PED, nx, ny, vz)
                        end
                    end
                    prev_pos = { x = px, y = py }
                end
            end
        else
            if anims_applied then
                -- возвращаем анимациям обычную скорость при выключении
                for _, anim in ipairs(SPEEDHACK_ANIMS) do
                    pcall(setCharAnimSpeed, PLAYER_PED, anim, 1.0)
                end
                anims_applied = false
            end
            prev_pos = nil
        end
    end
end)

-- ПОТОК: TELEPORT TO PLAYER
------------------------------------------------------------
-- телепорт к другому онлайн-игроку по SAMP ID; использует ту же надёжную
-- связку getAllChars + sampGetPlayerIdByCharHandle, подтверждённую рабочей
-- в ESP игроков ранее.
--
-- ВАЖНО: раньше эта функция вызывалась напрямую из клика кнопки внутри
-- imgui.OnFrame и содержала wait(100) — это запрещено (UI-кадр не является
-- корутиной lua_thread) и стабильно роняло скрипт ("Action crashed").
-- Теперь кнопка только ставит флаг-запрос (teleport_player_request), а сам
-- телепорт с wait() выполняется в отдельном lua_thread ниже.
--
-- Кроме того, мгновенный прыжок координат (setCharCoordinates за один тик)
-- на сотни метров — то, что чаще всего ловит анти-чит сервера как
-- невозможное перемещение и кикает. Поэтому телепорт выполняется плавным
-- "долётом" мелкими шагами, как и телепорт к метке.
local teleport_player_request = nil -- SAMP id игрока либо nil

local function resolve_player_target_coords(target_id)
    local p = samp_players[target_id]
    if p and p.x and not p.isNpc then
        local name = p.nickname or ('ID:' .. tostring(target_id))
        return p.x, p.y, p.z, name
    end

    -- если синк ещё не пришёл (игрок давно стоит) — пробуем через getAllChars как fallback
    local okAll, allChars = pcall(getAllChars)
    if okAll and allChars then
        for _, ped in ipairs(allChars) do
            if ped and ped ~= PLAYER_PED then
                local okId, isPlayer, sampId = pcall(sampGetPlayerIdByCharHandle, ped)
                if okId and isPlayer and sampId == target_id then
                    local okC, tx, ty, tz = pcall(getCharCoordinates, ped)
                    if okC and type(tx) == 'number' then
                        return tx, ty, tz, ('ID:' .. tostring(target_id))
                    end
                end
            end
        end
    end
    return nil
end

lua_thread.create(function()
    while true do
        wait(0)
        if teleport_player_request ~= nil then
            local target_id = teleport_player_request
            teleport_player_request = nil

            local tx, ty, tz, name = resolve_player_target_coords(target_id)
            if not tx then
                teleport_player_status = u8'Игрок не найден.'
            else
                pcall(requestCollision, tx, ty)
                pcall(loadScene, tx, ty, tz)

                local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
                if okP and type(px) == 'number' then
                    teleport_in_jumps(px, py, pz, tx, ty, tz + 0.5)
                    teleport_player_status = u8('Телепорт к ' .. name .. ' выполнен.')
                else
                    teleport_player_status = u8'Не удалось телепортироваться.'
                end
            end
        end
    end
end)

-- ПОТОК: NO CLIP
------------------------------------------------------------
-- Персонаж (не только камера) двигается сквозь любые препятствия, используя
-- СОБСТВЕННЫЙ виртуальный джойстик (noclip_stick) и свои кнопки высоты —
-- отдельные от Free Cam/Jetpack, чтобы функции не путали друг друга.
lua_thread.create(function()
    local was_enabled = false
    while true do
        wait(0)

        if player_noclip_enabled[0] then
            local okExistNC, existsNC = pcall(doesCharExist, PLAYER_PED)
            if not okExistNC or not existsNC then
                -- персонаж временно не существует (респавн/загрузка) — просто
                -- ждём следующего кадра, не трогаем координаты
                goto noclip_continue
            end

            if not was_enabled then
                local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
                if okP and type(px) == 'number' then
                    noclip_pos.x, noclip_pos.y, noclip_pos.z = px, py, pz
                end
                pcall(setCharProofs, PLAYER_PED, true, true, true, true, true)
                was_enabled = true
            end

            -- направление — по направлению текущей игровой камеры (свайп-поворот),
            -- только горизонтальная составляющая (без Z, чтобы не создавать
            -- паразитный подъём/спуск при наклоне камеры вверх-вниз)
            local okFwd, camFx, camFy = pcall(getActiveCameraPointAt)
            local okCamPos, camPx, camPy = pcall(getActiveCameraCoordinates)
            local fwdX, fwdY = 0, 1
            if okFwd and camFx and okCamPos and camPx then
                local dx, dy = camFx - camPx, camFy - camPy
                local len = math.sqrt(dx*dx + dy*dy)
                if len > 0 then fwdX, fwdY = dx/len, dy/len end
            end
            local rightX, rightY = fwdY, -fwdX

            local jx, jy = noclip_stick.x, noclip_stick.y
            local speed = NOCLIP_MOVE_SPEED
            if noclip_fast_mode then speed = speed * NOCLIP_FAST_MULT end

            noclip_pos.x = noclip_pos.x + (fwdX * (-jy) + rightX * jx) * speed
            noclip_pos.y = noclip_pos.y + (fwdY * (-jy) + rightY * jx) * speed

            if flight_vertical_dir ~= 0 then
                noclip_pos.z = noclip_pos.z + flight_vertical_dir * speed
            end

            pcall(setCharCoordinates, PLAYER_PED, noclip_pos.x, noclip_pos.y, noclip_pos.z)
        else
            if was_enabled then
                pcall(setCharProofs, PLAYER_PED, false, false, false, false, false)
                was_enabled = false
            end
        end
        ::noclip_continue::
    end
end)

-- ПОТОК: JETPACK
------------------------------------------------------------
-- Реактивный ранец: персонаж реально летает, управляется собственным
-- виртуальным джойстиком (свой, отдельный от Free Cam/No Clip, чтобы не
-- было залипаний общих кнопок между функциями). Горизонталь — джойстик,

local jetpack_move_up, jetpack_move_down = false, false

lua_thread.create(function()
    local was_enabled = false
    local jetpack_pos = { x = 0, y = 0, z = 0 }

    while true do
        wait(0)

        if player_jetpack_enabled[0] then
            local okExistJp, existsJp = pcall(doesCharExist, PLAYER_PED)
            if okExistJp and existsJp then
                if not was_enabled then
                    local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
                    if okP and type(px) == 'number' then
                        jetpack_pos.x, jetpack_pos.y, jetpack_pos.z = px, py, pz
                    end
                    was_enabled = true
                end

                local okFwd, camFx, camFy, camFz = pcall(getActiveCameraPointAt)
                local okCamPos, camPx, camPy, camPz = pcall(getActiveCameraCoordinates)
                local fwdX, fwdY = 0, 1
                if okFwd and camFx and okCamPos and camPx then
                    local dx, dy = camFx - camPx, camFy - camPy
                    local len = math.sqrt(dx*dx + dy*dy)
                    if len > 0 then fwdX, fwdY = dx/len, dy/len end
                end
                local rightX, rightY = fwdY, -fwdX

                local jx, jy = jetpack_stick.x, jetpack_stick.y
                local speed = JETPACK_MOVE_SPEED
                if jetpack_fast_mode then speed = speed * JETPACK_FAST_MULT end

                -- считаем целевое смещение ОТ текущей internal-позиции
                -- (jetpack_pos), а не перечитываем getCharCoordinates в этом
                -- же кадре — иначе накапливается ошибка, т.к. игра ещё не
                -- успела физически применить предыдущий setCharCoordinates
                jetpack_pos.x = jetpack_pos.x + (fwdX * (-jy) + rightX * jx) * speed
                jetpack_pos.y = jetpack_pos.y + (fwdY * (-jy) + rightY * jx) * speed

                if flight_vertical_dir ~= 0 then
                    jetpack_pos.z = jetpack_pos.z + flight_vertical_dir * speed
                end
                -- без ввода по вертикали — просто парим на текущей высоте,
                -- НЕ трогаем Z вообще (никаких перечитываний координат здесь)

                pcall(setCharCoordinates, PLAYER_PED, jetpack_pos.x, jetpack_pos.y, jetpack_pos.z)
                pcall(setCharProofs, PLAYER_PED, false, false, false, false, true) -- защита только от падения
            end
        elseif was_enabled then
            pcall(setCharProofs, PLAYER_PED, false, false, false, false, false)
            was_enabled = false
        end
    end
end)

-- РАЗДЕЛ RENDERING: HIDE PLAYERS (для FPS)
------------------------------------------------------------
-- ПЕРЕДЕЛАНО: старая версия использовала sampGetCharHandleBySampPlayerId для
-- поиска ped-хэндла по id — этот опкод на Arizona/Android (ARZMOD/MonetLoader)
-- сломан и всегда возвращает false (см. диагностику ESP выше: join=642,
-- но ни один char handle через этот опкод не находится). Именно поэтому
-- HIDE PLAYERS не работал вообще.
--
-- НОВЫЙ подход: id вообще не нужен. Берём хэндлы напрямую из getAllChars()
-- (это работает стабильно — на нём построен весь ESP) и переключаем
-- видимость/коллизию прямо на ped-хэндле: setCharVisible + setCharCollision.
-- Это не полный стрим-аут (движок всё ещё знает о персонаже), но даёт
-- заметный прирост FPS за счёт отключения отрисовки и физики коллизий,
-- и, в отличие от RPC-эмуляции, гарантированно работает и полностью
-- обратимо без реконнекта.
local render_hide_players_enabled = new.bool(false)
local hidden_players_peds = {} -- [ped] = true, если сейчас скрыт

lua_thread.create(function()
    while true do
        wait(300) -- толпа не меняется мгновенно, каждый кадр проверять незачем
        if render_hide_players_enabled[0] then
            local okAll, allChars = pcall(getAllChars)
            if okAll and allChars then
                local seen = {}
                for _, ped in ipairs(allChars) do
                    if ped and ped ~= PLAYER_PED then
                        local okExist, exists = pcall(doesCharExist, ped)
                        if okExist and exists then
                            seen[ped] = true
                            if not hidden_players_peds[ped] then
                                pcall(setCharVisible, ped, false)
                                pcall(setCharCollision, ped, false)
                                hidden_players_peds[ped] = true
                            end
                        end
                    end
                end
                -- чистим записи о ped'ах, которые перестали существовать
                -- (иначе таблица растёт бесконечно при смене состава игроков)
                for ped, _ in pairs(hidden_players_peds) do
                    if not seen[ped] then hidden_players_peds[ped] = nil end
                end
            end
        elseif next(hidden_players_peds) then
            -- восстанавливаем всех скрытых игроков при выключении функции
            for ped, _ in pairs(hidden_players_peds) do
                local okExist, exists = pcall(doesCharExist, ped)
                if okExist and exists then
                    pcall(setCharVisible, ped, true)
                    pcall(setCharCollision, ped, true)
                end
            end
            hidden_players_peds = {}
        end
    end
end)

-- РАЗДЕЛ RENDERING: HIDE VEHICLES (для FPS)

-- ПЕРЕДЕЛАНО по той же причине: sampGetVehicleIdByCarHandle — из той же
-- сломанной семьи опкодов SAMPFUNCS id-маппинга. Теперь работаем прямо
-- с vehicle-хэндлами из getAllVehicles() (нативно, всегда работает),
-- через setCarVisible/setCarCollision — полностью обратимо, в отличие
-- от старой RPC-выгрузки (raknetEmulRpcReceiveBitStream(165, ...)),
-- которая была необратима без реконнекта.
local render_hide_vehicles_enabled = new.bool(false)
local hidden_vehicles_handles = {} -- [veh] = true, если сейчас скрыт

local function is_vehicle_protected_handle(veh)
    -- своя машина (та, в которой сейчас едет игрок) никогда не скрывается
    local okExist, exists = pcall(doesCharExist, PLAYER_PED)
    if not okExist or not exists then return false end
    local okIn, inCar = pcall(isCharInAnyCar, PLAYER_PED)
    if okIn and inCar then
        local okMy, myVeh = pcall(storeCarCharIsInNoSave, PLAYER_PED)
        if okMy and myVeh == veh then return true end
    end
    return false
end

lua_thread.create(function()
    while true do
        wait(500) -- машины на карте меняются не так часто, чтобы проверять каждый кадр
        if render_hide_vehicles_enabled[0] then
            local okAll, allVehicles = pcall(getAllVehicles)
            if okAll and allVehicles then
                local seen = {}
                for _, veh in ipairs(allVehicles) do
                    local okExist, exists = pcall(doesVehicleExist, veh)
                    if okExist and exists then
                        seen[veh] = true
                        if not hidden_vehicles_handles[veh] and not is_vehicle_protected_handle(veh) then
                            pcall(setCarVisible, veh, false)
                            pcall(setCarCollision, veh, false)
                            hidden_vehicles_handles[veh] = true
                        end
                    end
                end
                for veh, _ in pairs(hidden_vehicles_handles) do
                    if not seen[veh] then hidden_vehicles_handles[veh] = nil end
                end
            end
        elseif next(hidden_vehicles_handles) then
            -- восстановление — теперь полностью обратимо без реконнекта
            for veh, _ in pairs(hidden_vehicles_handles) do
                local okExist, exists = pcall(doesVehicleExist, veh)
                if okExist and exists then
                    pcall(setCarVisible, veh, true)
                    pcall(setCarCollision, veh, true)
                end
            end
            hidden_vehicles_handles = {}
        end
    end
end)

------------------------------------------------------------
-- РАЗДЕЛ FPS: удаление декоративных объектов по категориям
------------------------------------------------------------
-- Источник данных и идеи — сторонний скрипт FPS UP (список ID моделей
-- по категориям + перехват RPC создания объекта). Перехватываем RPC 44
-- (CreateObject) и, если модель попадает в одну из включённых категорий,
-- отменяем создание объекта (return false из onReceiveRpc) — объект
-- вообще не появляется в мире клиента, а не просто становится невидимым.
-- Это дешевле по нагрузке, чем DestroyObject постфактум.
-- ВАЖНО: Lua-функция не может иметь больше 60 upvalue. UI-рендер AXION —
-- одна большая функция, и у неё уже много захваченных переменных. Поэтому
-- всё состояние раздела FPS хранится в ОДНОМ table (FPS.*), а не в
-- отдельных local-переменных — это даёт 1 upvalue вместо десятка.
local FPS = {
    objects_enabled = new.bool(false), -- общий рубильник REMOVE OBJECTS
    removed_objects_count = 0,
    category_toggles = {}, -- [имя категории] = new.bool(false)
    active_model_set = {}, -- быстрый набор моделей для включённых категорий

    distance_enabled = new.bool(false),
    nametags_distance = new.int(15),
    tdtext_distance = new.int(10),
    chatbubbles_distance = new.int(8),
    nametags_server_settings_ptr = nil,
    nametags_allowed_distance = nil,

    auto_unload_vehicles_enabled = new.bool(false),
    vehicle_unload_distance = new.int(80),
}
for _, name in ipairs(FPS_CATEGORY_ORDER) do
    FPS.category_toggles[name] = new.bool(false)
end

-- пересчитывается только при переключении чекбокса, а не на каждый объект
local function fps_rebuild_active_model_set()
    FPS.active_model_set = {}
    for _, name in ipairs(FPS_CATEGORY_ORDER) do
        if FPS.category_toggles[name][0] then
            local list = FPS_OBJECT_CATEGORIES[name]
            if list then
                for _, model in ipairs(list) do
                    FPS.active_model_set[model] = true
                end
            end
        end
    end
end

FPS.category_order = FPS_CATEGORY_ORDER
FPS.rebuild_active_model_set = fps_rebuild_active_model_set

function onReceiveRpc(id, bs)
    if id == 44 and FPS.objects_enabled[0] and sampIsLocalPlayerSpawned() then
        local okObjId, objId = pcall(raknetBitStreamReadInt16, bs)
        local okModel, model = pcall(raknetBitStreamReadInt32, bs)
        if okObjId and okModel and FPS.active_model_set[model] then
            FPS.removed_objects_count = FPS.removed_objects_count + 1
            return false -- отменяем создание объекта
        end
    end
end

-- АКТИВНОЕ СКАНИРОВАНИЕ уже существующих объектов -----------------------
-- Блокировка через onReceiveRpc выше действует только на НОВЫЕ объекты —
-- те, что были загружены/застримлены ДО включения REMOVE OBJECTS (или до
-- захода в новую зону), она не трогает, поэтому удаление казалось очень
-- медленным ("только при переезде по карте"). Этот поток догоняет их
-- активно: проходит по всем текущим объектам мира (getAllObjects) и сразу
-- удаляет те, что попадают в отмеченные категории.
--
-- ВАЖНО (фикс краша): на карте могут быть тысячи объектов. Обработка ВСЕХ
-- разом в одной итерации без wait() — классическая причина краша в
-- MoonLoader: скрипт слишком долго не отдаёт управление, и его убивает
-- watchdog. Именно это происходило при "Выбрать всё" (сразу ~5500 моделей
-- в фильтре -> тысячи совпадений разом). Поэтому обрабатываем ПОРЦИЯМИ
-- с частым wait(0) внутри цикла, а не одним блоком между wait(1000).
-- ЗНАЧИТЕЛЬНО безопаснее, чем было: маленькие порции (10 вместо 50) и
-- реальная пауза 15мс между ними (не wait(0), который отдаёт управление
-- лишь на долю кадра) — резкое удаление тысяч объектов разом (например,
-- сразу после "Выбрать всё" на весь список категорий) может крашить саму
-- игру, а не просто скрипт, если движок в этот момент ещё держит ссылки
-- на эти объекты (рендер/коллизия/стриминг). Плюс лимит удалений за один
-- проход цикла, чтобы никогда не сносить весь мир одним махом.
local FPS_SCAN_CHUNK_SIZE = 10   -- объектов за одну порцию перед yield
local FPS_SCAN_CHUNK_WAIT = 15   -- мс паузы между порциями
local FPS_MAX_DELETIONS_PER_PASS = 300 -- максимум удалений за один полный проход

lua_thread.create(function()
    while true do
        wait(1000)
        if FPS.objects_enabled[0] and next(FPS.active_model_set) then
            local okAll, allObjects = pcall(getAllObjects)
            if okAll and allObjects then
                local processed = 0
                local deletedThisPass = 0
                for _, obj in ipairs(allObjects) do
                    if deletedThisPass >= FPS_MAX_DELETIONS_PER_PASS then
                        break -- остаток доудалим на следующем проходе через 1с
                    end

                    local okExist, exists = pcall(doesObjectExist, obj)
                    if okExist and exists then
                        local okModel, model = pcall(getObjectModel, obj)
                        if okModel and FPS.active_model_set[model] then
                            local okDel = pcall(deleteObject, obj)
                            if okDel then
                                FPS.removed_objects_count = FPS.removed_objects_count + 1
                                deletedThisPass = deletedThisPass + 1
                            end
                        end
                    end
                    processed = processed + 1
                    if processed % FPS_SCAN_CHUNK_SIZE == 0 then
                        wait(FPS_SCAN_CHUNK_WAIT) -- реальная пауза, не одна доля кадра
                        -- если функцию выключили посреди сканирования — прерываемся
                        if not FPS.objects_enabled[0] then break end
                    end
                end
            end
        end
    end
end)

-- РАЗДЕЛ FPS: дистанция неймтегов / 3D-текста / чат-баблов
------------------------------------------------------------
-- Урезание дистанции прорисовки неймтегов, 3D-текстов и чат-пузырей
-- снижает число объектов, которые движок вообще пытается отрисовать
-- в толпе игроков. Работает через нативный указатель настроек сервера
-- (портируемо, в отличие от жёстко зашитых PC-адресов памяти) и через
-- перехват SAMP-событий создания 3D-текста/чат-бабла.
FPS.distance_enabled       = new.bool(false)
FPS.nametags_distance      = new.int(15)   -- метры, дефолт большинства серверов ~30-40
FPS.tdtext_distance        = new.int(10)
FPS.chatbubbles_distance   = new.int(8)

FPS.nametags_server_settings_ptr = nil
FPS.nametags_allowed_distance    = nil

local fps_nametags_diag_shown = false

local function fps_apply_nametags_distance()
    if not FPS.distance_enabled[0] then return end

    if not FPS.nametags_server_settings_ptr then
        local okPtr, ptr = pcall(sampGetServerSettingsPtr)
        if okPtr and ptr and ptr ~= 0 then
            FPS.nametags_server_settings_ptr = ptr + 39
            local okMem, memory = pcall(require, 'memory')
            if okMem then
                local okGet, val = pcall(memory.getfloat, FPS.nametags_server_settings_ptr)
                if okGet then FPS.nametags_allowed_distance = val end
            end
        elseif sampAddChatMessage and not fps_nametags_diag_shown then
            fps_nametags_diag_shown = true
            pcall(sampAddChatMessage,
                '{FF6347}[AXION] LIMIT DISTANCE: sampGetServerSettingsPtr недоступен на этой сборке — дистанция неймтегов игроков не будет урезана (3D-текст и чат-баблы работают отдельно и не затронуты).', -1)
        end
    end

    if FPS.nametags_server_settings_ptr then
        local okMem, memory = pcall(require, 'memory')
        if okMem then
            local target = FPS.nametags_distance[0]
            if FPS.nametags_allowed_distance and FPS.nametags_allowed_distance > 0
               and target > FPS.nametags_allowed_distance then
                target = FPS.nametags_allowed_distance
            end
            local okSet = pcall(memory.setfloat, FPS.nametags_server_settings_ptr, target)
            if not okSet and sampAddChatMessage and not fps_nametags_diag_shown then
                fps_nametags_diag_shown = true
                pcall(sampAddChatMessage,
                    '{FF6347}[AXION] LIMIT DISTANCE: запись в память не удалась на этой сборке.', -1)
            end
        end
    end
end

FPS.tdtext_events_fired = 0
FPS.chatbubble_events_fired = 0

function sampev.onCreate3DText(id, col, pos, allowed_dist, los, plid, vehid, text)
    if FPS.distance_enabled[0] then
        FPS.tdtext_events_fired = FPS.tdtext_events_fired + 1
        local custom = FPS.tdtext_distance[0]
        if custom < allowed_dist then
            return { id, col, pos, custom, los, plid, vehid, text }
        end
    end
end

function sampev.onPlayerChatBubble(id, col, allowed_dist, dur, text)
    if FPS.distance_enabled[0] then
        FPS.chatbubble_events_fired = FPS.chatbubble_events_fired + 1
        local custom = FPS.chatbubbles_distance[0]
        if custom < allowed_dist then
            return { id, col, custom, dur, text }
        end
    end
end

lua_thread.create(function()
    while true do
        wait(1000) -- дистанция не требует мгновенной реакции, раз в секунду достаточно
        if FPS.distance_enabled[0] then
            pcall(fps_apply_nametags_distance)
        end
    end
end)

------------------------------------------------------------
-- РАЗДЕЛ FPS: авто-скрытие дальних машин по дистанции
------------------------------------------------------------
-- ПЕРЕДЕЛАНО: раньше использовало sampGetVehicleIdByCarHandle (сломанный
-- опкод на Arizona/Android) + необратимую RPC-выгрузку. Теперь работает
-- прямо с vehicle-хэндлом (getAllVehicles — нативно, надёжно) и переиспользует
-- тот же механизм видимость+коллизия и ту же таблицу hidden_vehicles_handles,
-- что и HIDE VEHICLES из RENDERING — так они не конфликтуют между собой.
FPS.auto_unload_vehicles_enabled = new.bool(false)
FPS.vehicle_unload_distance      = new.int(80)

-- ключ отдельный от render_hide_vehicles, чтобы не путать источник скрытия:
-- если одну и ту же машину скрыли и HIDE VEHICLES, и AUTO UNLOAD, снятие
-- одной функции не должно возвращать видимость, если вторая всё ещё активна
local fps_hidden_by_unload = {} -- [veh] = true, если скрыт именно этой функцией

lua_thread.create(function()
    while true do
        wait(700)
        if FPS.auto_unload_vehicles_enabled[0] then
            local okPed, exists = pcall(doesCharExist, PLAYER_PED)
            if okPed and exists then
                local okPos, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
                if okPos then
                    local okAll, allVehicles = pcall(getAllVehicles)
                    if okAll and allVehicles then
                        local seen = {}
                        for _, veh in ipairs(allVehicles) do
                            local okExist, vExists = pcall(doesVehicleExist, veh)
                            if okExist and vExists then
                                seen[veh] = true
                                local okVPos, vx, vy, vz = pcall(getCarCoordinates, veh)
                                if okVPos then
                                    local dist = getDistanceBetweenCoords3d(px, py, pz, vx, vy, vz)
                                    local tooFar = dist > FPS.vehicle_unload_distance[0]

                                    if tooFar and not fps_hidden_by_unload[veh]
                                       and not is_vehicle_protected_handle(veh) then
                                        -- уехала далеко — скрываем
                                        pcall(setCarVisible, veh, false)
                                        pcall(setCarCollision, veh, false)
                                        fps_hidden_by_unload[veh] = true
                                        hidden_vehicles_handles[veh] = true
                                    elseif not tooFar and fps_hidden_by_unload[veh] then
                                        -- вернулись в радиус — восстанавливаем видимость,
                                        -- НО только если её не скрывает ещё и HIDE VEHICLES
                                        if not render_hide_vehicles_enabled[0] then
                                            pcall(setCarVisible, veh, true)
                                            pcall(setCarCollision, veh, true)
                                            hidden_vehicles_handles[veh] = nil
                                        end
                                        fps_hidden_by_unload[veh] = nil
                                    end
                                end
                            end
                        end
                        -- машина пропала из мира (выгружена стримингом) — забываем о ней,
                        -- чтобы таблица не росла бесконечно
                        for veh, _ in pairs(fps_hidden_by_unload) do
                            if not seen[veh] then fps_hidden_by_unload[veh] = nil end
                        end
                    end
                end
            end
        elseif next(fps_hidden_by_unload) then
            -- функцию выключили — возвращаем видимость всем машинам, которые
            -- скрыла именно она (если их не держит скрытыми ещё и HIDE VEHICLES)
            for veh, _ in pairs(fps_hidden_by_unload) do
                local okExist, exists = pcall(doesVehicleExist, veh)
                if okExist and exists and not render_hide_vehicles_enabled[0] then
                    pcall(setCarVisible, veh, true)
                    pcall(setCarCollision, veh, true)
                    hidden_vehicles_handles[veh] = nil
                end
            end
            fps_hidden_by_unload = {}
        end
    end
end)


-- из отдельного адреса памяти, а не только из игрового стата. Рабочий
-- способ (подтверждён рабочим сторонним скриптом) — нативная функция

            was_enabled = false
        end
    end
end)

------------------------------------------------------------
-- ПОТОК: FREE CAM
------------------------------------------------------------
-- Управление — СВОЙ виртуальный джойстик, нарисованный прямо в окне AXION
-- (см. UI ниже, переменная freecam_stick), а не игровой опкод — на этой
-- сборке игровые виджеты (WIDGET_PED_MOVE и т.п.) оказались недоступны,
-- поэтому рисуем собственный перетаскиваемый стик, который работает всегда.
-- Поворот камеры — обычные свайпы (используем текущее направление игровой
-- камеры, которое вы и так вращаете пальцем как всегда). Персонаж замирает
-- на месте, а его координаты, отправляемые на сервер, подменяются на
-- исходные, чтобы сервер не видел "телепорт" камеры.

lua_thread.create(function()
    local was_enabled = false

    while true do
        wait(0)

        if player_freecam_enabled[0] then
            if not was_enabled then
                local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
                if okP and type(px) == 'number' then
                    freecam_frozen_pos = { x = px, y = py, z = pz }
                    freecam_pos.x, freecam_pos.y, freecam_pos.z = px, py, pz + 1.0
                end

                -- берём текущее направление игровой камеры как стартовое
                local okFwd0, fx0, fy0 = pcall(getActiveCameraPointAt)
                local okCam0, cx0, cy0 = pcall(getActiveCameraCoordinates)
                if okFwd0 and okCam0 and fx0 and cx0 then
                    freecam_yaw = math.deg(math.atan(fx0 - cx0, fy0 - cy0))
                else
                    freecam_yaw = 0
                end
                freecam_pitch = 0

                pcall(freezeCharPosition, PLAYER_PED, true)
                was_enabled = true
            end

            -- держим персонажа замороженным на зафиксированной точке
            if freecam_frozen_pos then
                pcall(setCharCoordinates, PLAYER_PED,
                    freecam_frozen_pos.x, freecam_frozen_pos.y, freecam_frozen_pos.z)
            end

            -- ПОВОРОТ: свайпы — используем разницу текущего направления
            -- игровой камеры относительно предыдущего кадра (пользователь
            -- крутит её пальцем как обычно), просто переносим эту дельту
            -- на угол полёта freecam.
            local okFwd, camFx, camFy, camFz = pcall(getActiveCameraPointAt)
            local okCamPos, camPx, camPy, camPz = pcall(getActiveCameraCoordinates)
            if okFwd and camFx and okCamPos and camPx then
                local dx, dy, dz = camFx - camPx, camFy - camPy, camFz - camPz
                local lenXY = math.sqrt(dx*dx + dy*dy)
                local len   = math.sqrt(dx*dx + dy*dy + dz*dz)
                if len > 0 then
                    freecam_yaw   = math.deg(math.atan(dx, dy))
                    freecam_pitch = math.deg(math.atan(dz, math.max(lenXY, 0.0001)))
                end
            end

            -- ДВИЖЕНИЕ: собственный виртуальный джойстик из UI
            local jx, jy = freecam_stick.x, freecam_stick.y
            local speed = FREECAM_MOVE_SPEED
            if freecam_fast_mode then speed = speed * FREECAM_FAST_MULT end

            local yawR = math.rad(freecam_yaw)
            local fwdX, fwdY = math.sin(yawR), math.cos(yawR)
            local rightX, rightY = math.cos(yawR), -math.sin(yawR)

            -- jy < 0 обычно значит "вперёд" на большинстве джойстиков движения
            freecam_pos.x = freecam_pos.x + (fwdX * (-jy) + rightX * jx) * speed
            freecam_pos.y = freecam_pos.y + (fwdY * (-jy) + rightY * jx) * speed

            -- отдельные кнопки вверх/вниз (зажатие) остаются для вертикали,
            -- так как джойстик движения не даёт высоты
            if flight_vertical_dir ~= 0 then
                freecam_pos.z = freecam_pos.z + flight_vertical_dir * speed
            end

            -- применяем позицию и направление взгляда камеры
            pcall(setFixedCameraPosition, freecam_pos.x, freecam_pos.y, freecam_pos.z, 0, 0, 0)
            local pitchR = math.rad(freecam_pitch)
            local lookDist = 10
            local lookX = freecam_pos.x + fwdX * math.cos(pitchR) * lookDist
            local lookY = freecam_pos.y + fwdY * math.cos(pitchR) * lookDist
            local lookZ = freecam_pos.z + math.sin(pitchR) * lookDist
            pcall(pointCameraAtPoint, lookX, lookY, lookZ, 2)

        else
            if was_enabled then
                pcall(freezeCharPosition, PLAYER_PED, false)
                pcall(restoreCameraJumpcut)
                freecam_frozen_pos = nil
                was_enabled = false
            end
        end
    end
end)

-- Подменяем координаты в исходящем пакете синхронизации, пока FreeCam включён,
-- чтобы сервер продолжал видеть игрока на замороженной точке, а не считал,
-- что произошёл телепорт/спидхак из-за перемещения камеры.
if sampev.onSendPlayerSync == nil then
    function sampev.onSendPlayerSync(data)
        if player_freecam_enabled[0] and freecam_frozen_pos and data and data.position then
            data.position.x = freecam_frozen_pos.x
            data.position.y = freecam_frozen_pos.y
            data.position.z = freecam_frozen_pos.z
            return data
        end
    end
else
    local prev_onSendPlayerSync = sampev.onSendPlayerSync
    function sampev.onSendPlayerSync(data)
        if player_freecam_enabled[0] and freecam_frozen_pos and data and data.position then
            data.position.x = freecam_frozen_pos.x
            data.position.y = freecam_frozen_pos.y
            data.position.z = freecam_frozen_pos.z
        end
        return prev_onSendPlayerSync(data)
    end
end

-- рисует виртуальный джойстик (перетаскиваемая ручка в круге) и записывает
-- значение в stick_ref.x/y ([-1..1]); id_str должен быть уникальным для
-- каждого места использования (Free Cam/No Clip/Jetpack — разные джойстики)
local function draw_virtual_stick(id_str, stick_ref)
    local STICK_SIZE  = 160
    local KNOB_RADIUS = 28
    local avail_w = imgui.GetWindowWidth() - imgui.GetStyle().WindowPadding.x * 2
    local stick_cursor = imgui.GetCursorScreenPos()
    local center_x = stick_cursor.x + avail_w / 2
    local center_y = stick_cursor.y + STICK_SIZE / 2

    imgui.InvisibleButton(u8('##stick_' .. id_str), imgui.ImVec2(avail_w, STICK_SIZE))
    local stick_active = imgui.IsItemActive()

    local draw_list = imgui.GetWindowDrawList()
    draw_list:AddCircleFilled(imgui.ImVec2(center_x, center_y), STICK_SIZE / 2,
        imgui.ColorConvertFloat4ToU32(imgui.ImVec4(0.12, 0.10, 0.18, 1.0)))
    draw_list:AddCircle(imgui.ImVec2(center_x, center_y), STICK_SIZE / 2,
        imgui.ColorConvertFloat4ToU32(imgui.ImVec4(ACCENT[1], ACCENT[2], ACCENT[3], 0.6)), 0, 2.0)

    if stick_active then
        local mp = imgui.GetMousePos()
        local dx, dy = mp.x - center_x, mp.y - center_y
        local dist = math.sqrt(dx*dx + dy*dy)
        local maxDist = STICK_SIZE / 2 - KNOB_RADIUS / 2
        if dist > maxDist and dist > 0 then
            dx, dy = dx / dist * maxDist, dy / dist * maxDist
        end
        stick_ref.x = dx / maxDist
        stick_ref.y = dy / maxDist

        local knob_x, knob_y = center_x + dx, center_y + dy
        draw_list:AddCircleFilled(imgui.ImVec2(knob_x, knob_y), KNOB_RADIUS / 2,
            imgui.ColorConvertFloat4ToU32(imgui.ImVec4(ACCENT[1], ACCENT[2], ACCENT[3], 1.0)))
    else
        -- отпустили — стик возвращается в центр, движение прекращается
        stick_ref.x, stick_ref.y = 0, 0
        draw_list:AddCircleFilled(imgui.ImVec2(center_x, center_y), KNOB_RADIUS / 2,
            imgui.ColorConvertFloat4ToU32(imgui.ImVec4(0.5, 0.45, 0.6, 0.8)))
    end

    imgui.Dummy(imgui.ImVec2(0, STICK_SIZE + 6))
end

-- Отдельное окно без рамки/заголовка, позиционированное там же, где обычно
-- находится игровой тач-стик движения (нижний левый угол экрана) — так
-- управление полётом не отвлекает от остального интерфейса и ощущается
-- как обычный игровой джойстик. Показывается только когда активна одна
-- из трёх функций полёта.
--
-- Логика зон:
--  • ВНУТРИ круга — обычный джойстик движения (x = вправо/влево, y = вперёд/назад)
--  • Вертикальная полоса СПРАВА от круга — свайп вверх/вниз меняет высоту
--    полёта (соответствует привычному "потянул вверх — взлетел")

-- Размеры элементов управления полётом
local FLIGHT_STICK_SIZE = 180   -- диаметр круга джойстика
local FLIGHT_VBAR_WIDTH = 52    -- ширина вертикальной полосы высоты
local FLIGHT_VBAR_GAP   = 16    -- отступ между джойстиком и полосой

-- Чувствительность поворота камеры: сколько градусов/сек при максимальном отклонении
local FLIGHT_CAM_SENS = 90.0

-- Накопленная дельта поворота камеры за кадр (применяется отдельным потоком)
local flight_cam_yaw_delta = 0.0

local function get_active_flight_stick()
    if player_freecam_enabled[0] then return freecam_stick end
    if player_noclip_enabled[0]  then return noclip_stick end
    if player_jetpack_enabled[0] then return jetpack_stick end
    return nil
end

-- Поток: применяем поворот камеры через heading персонажа.
-- На Android/MonetLoader setCharHeading — единственный надёжный способ.
lua_thread.create(function()
    while true do
        wait(0)
        local is_flight = player_freecam_enabled[0] or player_noclip_enabled[0] or player_jetpack_enabled[0]
        if is_flight and math.abs(flight_cam_yaw_delta) > 0.01 then
            local okH, heading = pcall(getCharHeading, PLAYER_PED)
            if okH and type(heading) == 'number' then
                local newH = (heading + flight_cam_yaw_delta) % 360
                pcall(setCharHeading, PLAYER_PED, newH)
            end
            flight_cam_yaw_delta = 0.0
        end
    end
end)

imgui.OnFrame(
    function()
        return player_freecam_enabled[0] or player_noclip_enabled[0] or player_jetpack_enabled[0]
    end,
    function()
        local stick_ref = get_active_flight_stick()
        if not stick_ref then return end

        local screen_w, screen_h = 1280, 720
        local okIO, io_ref = pcall(imgui.GetIO)
        if okIO and io_ref and io_ref.DisplaySize then
            screen_w, screen_h = io_ref.DisplaySize.x, io_ref.DisplaySize.y
        end

        local margin     = 26
        local total_w    = FLIGHT_STICK_SIZE + FLIGHT_VBAR_GAP + FLIGHT_VBAR_WIDTH
        local win_y      = screen_h - FLIGHT_STICK_SIZE - margin - 40

        -- ЛЕВАЯ панель: джойстик
        local left_x = margin
        imgui.SetNextWindowPos(imgui.ImVec2(left_x, win_y), imgui.Cond.Always)
        imgui.SetNextWindowSize(imgui.ImVec2(FLIGHT_STICK_SIZE, FLIGHT_STICK_SIZE), imgui.Cond.Always)
        pcall(imgui.SetNextWindowBgAlpha, 0)

        local function safe_flag(name)
            local ok, val = pcall(function() return imgui.WindowFlags[name] end)
            if ok and val then return val end
            return 0
        end
        local base_flags = safe_flag('NoDecoration') + safe_flag('NoMove')
            + safe_flag('NoScrollWithMouse') + safe_flag('NoSavedSettings')
            + safe_flag('NoFocusOnAppearing') + safe_flag('NoNav')
            + safe_flag('NoBackground')

        imgui.Begin(u8'##flight_left_window', nil, base_flags)
        local dl_left  = imgui.GetWindowDrawList()
        local lpos     = imgui.GetWindowPos()

        local col_accent  = imgui.ColorConvertFloat4ToU32(imgui.ImVec4(ACCENT[1], ACCENT[2], ACCENT[3], 0.75))
        local col_bg      = imgui.ColorConvertFloat4ToU32(imgui.ImVec4(0.10, 0.09, 0.15, 0.55))
        local col_knob    = imgui.ColorConvertFloat4ToU32(imgui.ImVec4(ACCENT[1], ACCENT[2], ACCENT[3], 1.00))
        local col_idle    = imgui.ColorConvertFloat4ToU32(imgui.ImVec4(0.55, 0.50, 0.65, 0.75))
        local col_hint    = imgui.ColorConvertFloat4ToU32(imgui.ImVec4(1, 1, 1, 0.50))

        -- Круг джойстика
        local cx = lpos.x + FLIGHT_STICK_SIZE / 2
        local cy = lpos.y + FLIGHT_STICK_SIZE / 2
        local radius  = FLIGHT_STICK_SIZE / 2 - 4
        local knob_r  = 26

        dl_left:AddCircleFilled(imgui.ImVec2(cx, cy), radius, col_bg)
        dl_left:AddCircle(imgui.ImVec2(cx, cy), radius, col_accent, 0, 2.5)
        -- крестик-подсказка направлений
        dl_left:AddText(imgui.ImVec2(cx - 4, lpos.y + 6),        col_hint, u8'^')
        dl_left:AddText(imgui.ImVec2(cx - 4, lpos.y + FLIGHT_STICK_SIZE - 22), col_hint, u8'v')
        dl_left:AddText(imgui.ImVec2(lpos.x + 6, cy - 7),        col_hint, u8'<')
        dl_left:AddText(imgui.ImVec2(lpos.x + FLIGHT_STICK_SIZE - 16, cy - 7), col_hint, u8'>')

        imgui.SetCursorScreenPos(imgui.ImVec2(lpos.x, lpos.y))
        imgui.InvisibleButton(u8'##flight_joystick', imgui.ImVec2(FLIGHT_STICK_SIZE, FLIGHT_STICK_SIZE))
        local joy_active = imgui.IsItemActive()

        if joy_active then
            local mp   = imgui.GetMousePos()
            local dx   = mp.x - cx
            local dy   = mp.y - cy
            local dist = math.sqrt(dx*dx + dy*dy)
            local maxD = radius - knob_r / 2
            if dist > maxD and dist > 0 then
                dx, dy = dx / dist * maxD, dy / dist * maxD
            end

            stick_ref.x = dx / maxD   -- вправо/влево: движение
            stick_ref.y = dy / maxD   -- вперёд/назад: движение

            -- Поворот камеры пропорционален горизонтальному отклонению джойстика.
            -- ~60 fps → delta per frame = sens * normalized_x / 60
            flight_cam_yaw_delta = -stick_ref.x * FLIGHT_CAM_SENS / 60.0

            dl_left:AddCircleFilled(imgui.ImVec2(cx + dx, cy + dy), knob_r, col_knob)
        else
            stick_ref.x, stick_ref.y = 0, 0
            flight_cam_yaw_delta = 0.0
            dl_left:AddCircleFilled(imgui.ImVec2(cx, cy), knob_r, col_idle)
        end

        imgui.End()

        -- ПРАВАЯ панель: вертикальная полоса высоты
        local right_x = screen_w - FLIGHT_VBAR_WIDTH - margin
        imgui.SetNextWindowPos(imgui.ImVec2(right_x, win_y), imgui.Cond.Always)
        imgui.SetNextWindowSize(imgui.ImVec2(FLIGHT_VBAR_WIDTH, FLIGHT_STICK_SIZE), imgui.Cond.Always)
        pcall(imgui.SetNextWindowBgAlpha, 0)

        imgui.Begin(u8'##flight_right_window', nil, base_flags)
        local dl_right = imgui.GetWindowDrawList()
        local rpos     = imgui.GetWindowPos()

        local vbar_cx  = rpos.x + FLIGHT_VBAR_WIDTH / 2
        local vbar_cy  = rpos.y + FLIGHT_STICK_SIZE / 2

        dl_right:AddRectFilled(
            imgui.ImVec2(rpos.x, rpos.y),
            imgui.ImVec2(rpos.x + FLIGHT_VBAR_WIDTH, rpos.y + FLIGHT_STICK_SIZE),
            col_bg, 10.0)
        dl_right:AddRect(
            imgui.ImVec2(rpos.x, rpos.y),
            imgui.ImVec2(rpos.x + FLIGHT_VBAR_WIDTH, rpos.y + FLIGHT_STICK_SIZE),
            col_accent, 10.0, nil, 2.0)
        dl_right:AddText(imgui.ImVec2(vbar_cx - 4, rpos.y + 5),                   col_hint, u8'^')
        dl_right:AddText(imgui.ImVec2(vbar_cx - 4, rpos.y + FLIGHT_STICK_SIZE - 20), col_hint, u8'v')

        imgui.SetCursorScreenPos(imgui.ImVec2(rpos.x, rpos.y))
        imgui.InvisibleButton(u8'##flight_vbar', imgui.ImVec2(FLIGHT_VBAR_WIDTH, FLIGHT_STICK_SIZE))
        local vbar_active = imgui.IsItemActive()

        if vbar_active then
            local mp  = imgui.GetMousePos()
            local rel = (mp.y - vbar_cy) / (FLIGHT_STICK_SIZE / 2)
            if rel < -0.15 then
                flight_vertical_dir = 1
            elseif rel > 0.15 then
                flight_vertical_dir = -1
            else
                flight_vertical_dir = 0
            end
            local ky = vbar_cy + math.max(-1, math.min(1, rel)) * (FLIGHT_STICK_SIZE / 2 - 16)
            dl_right:AddCircleFilled(imgui.ImVec2(vbar_cx, ky), 14, col_knob)
        else
            flight_vertical_dir = 0
            dl_right:AddCircleFilled(imgui.ImVec2(vbar_cx, vbar_cy), 14, col_idle)
        end

        imgui.End()
    end
)

-- ОТРИСОВКА ESP (читает только готовый кэш, ничего тяжёлого в кадре)

-- пересчитывает мировую точку в экранную и сообщает, находится ли она перед камерой
-- (convert3DCoordsToScreen может вернуть координаты "позади" камеры как валидные,
-- поэтому дополнительно проверяем через getCamMatrix проекцию на forward-вектор)
local function worldToScreenSigned(x, y, z, camPos, camFwd)
    local okS, sx, sy = pcall(convert3DCoordsToScreen, x, y, z)
    if not okS or not sx then return nil end

    local dx, dy, dz = x - camPos[1], y - camPos[2], z - camPos[3]
    local dot = dx * camFwd[1] + dy * camFwd[2] + dz * camFwd[3]

    return sx, sy, dot > 0 -- true = точка перед камерой
end

-- обрезает луч (cx,cy) -> (tx,ty) по границе прямоугольника экрана, возвращает точку на краю
local function clipToScreenEdge(cx, cy, tx, ty, sw, sh, margin)
    margin = margin or 20
    local minX, minY = margin, margin
    local maxX, maxY = sw - margin, sh - margin

    local dx, dy = tx - cx, ty - cy
    if dx == 0 and dy == 0 then return cx, cy end

    local tMin = math.huge

    if dx ~= 0 then
        local t1 = (minX - cx) / dx
        local t2 = (maxX - cx) / dx
        if t1 > 0 then tMin = math.min(tMin, t1) end
        if t2 > 0 then tMin = math.min(tMin, t2) end
    end
    if dy ~= 0 then
        local t3 = (minY - cy) / dy
        local t4 = (maxY - cy) / dy
        if t3 > 0 then tMin = math.min(tMin, t3) end
        if t4 > 0 then tMin = math.min(tMin, t4) end
    end

    if tMin == math.huge then return cx, cy end

    return cx + dx * tMin, cy + dy * tMin
end

-- единая функция отрисовки одного кэша
local function draw_esp_cache(draw_list, cache, psx, psy, sw, sh, camPos, camFwd,
                               lines_enabled, box_enabled, lines_size, color, show_name)
    for _, obj in ipairs(cache) do
        local sx, sy, inFront = nil, nil, true

        if camPos and camFwd then
            sx, sy, inFront = worldToScreenSigned(obj.x, obj.y, obj.z, camPos, camFwd)
        else
            local okS, rx, ry = pcall(convert3DCoordsToScreen, obj.x, obj.y, obj.z)
            if okS then sx, sy = rx, ry end
        end

        if sx then
            if inFront then
                if box_enabled[0] then
                    -- bounding box вокруг персонажа: высота человека ~2.1м в игровых
                    -- единицах, масштабируем размер прямоугольника по дистанции,
                    -- проецируя верхнюю и нижнюю точку модели на экран.
                    local okTop, tsx, tsy = pcall(convert3DCoordsToScreen, obj.x, obj.y, obj.z + 1.05)
                    local okBot, bsx, bsy = pcall(convert3DCoordsToScreen, obj.x, obj.y, obj.z - 1.05)
                    if okTop and okBot and tsx and bsx then
                        local boxHeight = math.abs(bsy - tsy)
                        local boxWidth  = boxHeight * 0.55
                        local left  = sx - boxWidth / 2
                        local right = sx + boxWidth / 2
                        local top   = tsy
                        local bot   = bsy
                        draw_list:AddRect(
                            imgui.ImVec2(left, top), imgui.ImVec2(right, bot),
                            color, 2.0, nil, lines_size[0])
                    else
                        draw_list:AddCircleFilled(imgui.ImVec2(sx, sy), 4.5, color)
                    end
                end
                if show_name and obj.name then
                    draw_list:AddText(imgui.ImVec2(sx + 6, sy - 7), color, obj.name)
                end
                if lines_enabled[0] then
                    draw_list:AddLine(
                        imgui.ImVec2(psx, psy), imgui.ImVec2(sx, sy),
                        color, lines_size[0])
                end
            elseif lines_enabled[0] then
                local dirX = psx - (sx - psx)
                local dirY = psy - (sy - psy)
                local ex, ey = clipToScreenEdge(psx, psy, dirX, dirY, sw, sh)
                draw_list:AddLine(
                    imgui.ImVec2(psx, psy), imgui.ImVec2(ex, ey),
                    color, lines_size[0])
            end
        end
    end
end

imgui.OnFrame(
    function()
        return (esp_enabled[0] and #esp_object_cache > 0)
            or (esp_players_enabled[0] and #esp_players_cache > 0)
            or (esp_cars_enabled[0]    and #esp_cars_cache    > 0)
    end,
    function()
        local okDl, draw_list = pcall(imgui.GetForegroundDrawList)
        if not okDl or not draw_list then return end

        local okP, px, py, pz = pcall(getCharCoordinates, PLAYER_PED)
        if not okP or type(px) ~= 'number' then return end

        local okScr, psx, psy = pcall(convert3DCoordsToScreen, px, py, pz)
        if not okScr or not psx then return end

        local okRes, sw, sh = pcall(getScreenResolution)
        if not okRes or not sw then sw, sh = 1280, 720 end

        local okCam, camX, camY, camZ = pcall(getActiveCameraCoordinates)
        local okFwd, fwdX, fwdY, fwdZ = pcall(getActiveCameraPointAt)
        local camPos, camFwd
        if okCam and camX and okFwd and fwdX then
            camPos = { camX, camY, camZ }
            local vx, vy, vz = fwdX - camX, fwdY - camY, fwdZ - camZ
            local len = math.sqrt(vx*vx + vy*vy + vz*vz)
            if len > 0 then camFwd = { vx/len, vy/len, vz/len } end
        end

        -- OBJECTS
        if esp_enabled[0] and #esp_object_cache > 0 then
            local col = imgui.ColorConvertFloat4ToU32(
                imgui.ImVec4(esp_lines_color[0], esp_lines_color[1], esp_lines_color[2], esp_lines_color[3]))
            draw_esp_cache(draw_list, esp_object_cache, psx, psy, sw, sh, camPos, camFwd,
                esp_lines_enabled, esp_dot_enabled, esp_lines_size, col, false)
        end

        -- PLAYERS
        if esp_players_enabled[0] and #esp_players_cache > 0 then
            local col = imgui.ColorConvertFloat4ToU32(
                imgui.ImVec4(esp_players_color[0], esp_players_color[1], esp_players_color[2], esp_players_color[3]))
            draw_esp_cache(draw_list, esp_players_cache, psx, psy, sw, sh, camPos, camFwd,
                esp_players_lines_enabled, esp_players_box_enabled, esp_players_lines_size, col, true)
        end

        -- CARS
        if esp_cars_enabled[0] and #esp_cars_cache > 0 then
            local col = imgui.ColorConvertFloat4ToU32(
                imgui.ImVec4(esp_cars_color[0], esp_cars_color[1], esp_cars_color[2], esp_cars_color[3]))
            draw_esp_cache(draw_list, esp_cars_cache, psx, psy, sw, sh, camPos, camFwd,
                esp_cars_lines_enabled, esp_cars_box_enabled, esp_cars_lines_size, col, false)
        end
    end
)
